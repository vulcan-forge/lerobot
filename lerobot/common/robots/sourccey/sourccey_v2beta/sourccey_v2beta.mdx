--- Sourccey V2 Beta Commands ---

```
python -m lerobot.calibrate \
    --robot.type=sourccey_v2beta \
    --robot.id=sourccey_v2beta
```

To teleoperate, SSH into your Raspberry Pi, and run `conda activate lerobot` and this command:

```bash
python -m lerobot.common.robots.sourccey.sourccey_v2beta.sourccey_v2beta_host
```

--- ACT Training

```bash
python lerobot/scripts/train.py \
  --dataset.repo_id=local/sourccey_v2beta_towel_010_a \
  --policy.type=act \
  --output_dir=outputs/train/act_sourccey_v2beta_towel_010_a \
  --job_name=act_sourccey_v2beta_towel_010_a \
  --policy.device=cuda \
  --policy.use_amp=true \
  --wandb.enable=false \
  --steps=100000
```

```bash
torchrun --nproc_per_node=2 lerobot/scripts/train.py \
  --dataset.repo_id=local/sourccey_v2beta_towel_010_a  \
  --policy.type=act \
  --output_dir=outputs/train/act_sourccey_v2beta_towel_010_a \
  --job_name=act_sourccey_v2beta_towel_010_a  \
  --policy.device=cuda \
  --policy.use_amp=true \
  --wandb.enable=false \
  --steps=100000 \
  --distributed_training=true \
  --num_gpus=2
```

---- Train on Pretrained SmolVLA model

```bash

 --dataset.repo_id=local/sourccey_v2beta_towel_010_a \
 --policy.path=lerobot/smolvla_base \
 --policy.device=cuda \
```
