```bash

    --robot.type=so100_follower \

```

```
python -m lerobot.teleoperate \
    --robot.type=so100_follower \
    --robot.port=COM24 \
    --robot.id=so100 \
    --teleop.type=so100_leader \
    --teleop.port=COM26 \
    --teleop.id=so100
```

```
python -m lerobot.record \
    --robot.type=so100_follower \
    --robot.port=COM24 \
    --robot.id=so100 \
    --teleop.type=so100_leader \
    --teleop.port=COM26 \
    --teleop.id=so100 \
    --display_data=true \
    --dataset.repo_id=local/so100_001_a \
    --dataset.num_episodes=1 \
    --dataset.single_task="Move around" \
    --dataset.push_to_hub=false
```
